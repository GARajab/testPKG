    const net = require('net');
const http = require('http');
const fs = require('fs');

const PC_IP = "192.168.1.100";
const CALLBACK_PORT = 9022;
const NPX_PORT = 8080;

class CallbackServer {
    constructor() {
        this.pkgSize = 0;
        this.pkgName = "";
        this.callbackSocket = null;
    }

    async start() {
        console.log("üîÑ Starting callback server...");
        
        const server = net.createServer((socket) => {
            this.callbackSocket = socket;
            console.log(`‚úÖ PS4 connected from ${socket.remoteAddress}`);
            this.sendMetadata();
            
            // Start streaming the package
            this.streamPackage();
        });

        server.listen(CALLBACK_PORT, () => {
            console.log(`üì° Callback server listening on ${PC_IP}:${CALLBACK_PORT}`);
        });

        server.on("error", (err) => {
            console.error(`‚ùå Server error: ${err.message}`);
        });
    }

    sendMetadata() {
        if (!this.callbackSocket) return;
        
        try {
            // Decode metadata from base64
            const metadata = Buffer.from("PASTE_METADATA_HERE", 'base64');
            this.callbackSocket.write(metadata);
            console.log("‚úÖ Metadata sent to PS4");
        } catch (err) {
            console.error(`‚ùå Metadata error: ${err.message}`);
        }
    }

    async streamPackage() {
        const pkgUrl = `http://${PC_IP}:${NPX_PORT}/pkgs/${this.pkgName}`;
        console.log(`üì• Streaming from: ${pkgUrl}`);

        http.get(pkgUrl, (res) => {
            let sentBytes = 0;
            let lastReport = 0;

            res.on("data", (chunk) => {
                if (this.callbackSocket) {
                    this.callbackSocket.write(chunk);
                    sentBytes += chunk.length;
                    
                    if (sentBytes - lastReport >= 5 * 1024 * 1024) {
                        const progress = (sentBytes / this.pkgSize) * 100;
                        console.log(`üìä ${progress.toFixed(1)}% (${sentBytes}/${this.pkgSize} bytes)`);
                        lastReport = sentBytes;
                    }
                }
            });

            res.on("end", () => {
                console.log(`‚úÖ Transfer complete: ${sentBytes} bytes`);
                if (this.callbackSocket) this.callbackSocket.end();
            });

            res.on("error", (err) => {
                console.error(`‚ùå Stream error: ${err.message}`);
                if (this.callbackSocket) this.callbackSocket.destroy();
            });
        });
    }
}

// Get package name from command line
const pkgName = process.argv[2];
if (!pkgName) {
    console.log("Usage: node callback-server.js ");
    process.exit(1);
}

const server = new CallbackServer();
server.pkgName = pkgName;
server.pkgSize = parseInt(process.argv[3]) || 0;
server.start();
